inner join 
================================================================================
select s.*,p.* from staff as s inner join payment as p where s.staff_id=p.staff_id;
-----------------------------------------------------------------------------------------------

left join
================================================================================
select s.*,p.* from staff as s left join payment as p on  s.staff_id=p.staff_id;
-----------------------------------------------------------------------------------------------

right join
================================================================================
select s.*,p.* from staff as s right join payment as p on  s.staff_id=p.staff_id;
-----------------------------------------------------------------------------------------------
